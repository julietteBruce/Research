Graphs :: newDigraph

newDigraph -- key used in the output of topSort

Description

This is a key of the hashtable output SortedDigraph of topSort.

i1 : G = digraph{{5,2},{5,0},{4,0},{4,1},{2,3},{3,1}}

o1 = Digraph{0 => {}    }
             1 => {}
             2 => {3}
             3 => {1}
             4 => {0, 1}
             5 => {2, 0}

o1 : Digraph
 
i2 : H = topSort G

o2 = SortedDigraph{digraph => Digraph{0 => {}    }   }
                                      1 => {}
                                      2 => {3}
                                      3 => {1}
                                      4 => {0, 1}
                                      5 => {2, 0}
                   map => HashTable{0 => 4}
                                    1 => 6
                                    2 => 3
                                    3 => 5
                                    4 => 2
                                    5 => 1
                   newDigraph => Digraph{1 => {3, 4}}
                                         2 => {4, 6}
                                         3 => {5}
                                         4 => {}
                                         5 => {6}
                                         6 => {}

o2 : SortedDigraph
 
i3 : keys H

o3 = {map, newDigraph, digraph}

o3 : List
