Example from Schubert: Euler characteristic of Horrocks-Mumford bundle

> proj(4,h,tang):            # need tangentbundle for chi
i1 : X = abstractProjectiveSpace'(4,base n,VariableName => h)

o1 = X

o1 : a flag bundle with subquotient ranks {4, 1}
 
> F:=sheaf(2,[5*h,10*h^2]):  # defines the Horrocks-Mumford bundle
i2 : F = abstractSheaf(X, Rank => 2, ChernClass => 1 + 5*h + 10*h^2)

o2 = F

o2 : an abstract sheaf of rank 2 on X
 
> chi(F&*o(n*h));            # computes chi of its twists
                          4        3   125  2
                    1/12 n  + 5/3 n  + --- n  + 125/6 n + 2
                                        12
i3 : chi (F ** OO(n*h))

      1 4   5 3   125 2   125
o3 = --n  + -n  + ---n  + ---n + 2
     12     3      12      6

o3 : QQ[n]
